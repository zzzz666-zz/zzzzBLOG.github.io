    create database `database`;  # ''是为了和关键字区分，创建一个名字为database的数据库
    show databases; # 显示所有数据库
    drop database `database`; # 删除名字叫database的数据库

    CREATE DATABASE `bzh678`; # ``是为了和关键字区分，创建名字叫bzh678的数据库
    SHOW DATABASES; # 展示所有的数据库
    DROP DATABASE `bzh678`; # 删除名字叫bzh678的数据库
    
    CREATE DATABASE `sql_tutorial`;
    USE `sql_tutorial`;
    
    # 创建一个table，名字叫student，有三个属性student_id、name、major。student_id是主键 int类型，name字符串类型，major 字符串类型
    CREATE TABLE `student`(
        `student_id` INT PRIMARY KEY AUTO_INCREMENT,
        `name` VARCHAR(50) NOT NULL, # 约束 不为空
        `major` VARCHAR(50) UNIQUE # 不可重复
    );
    
    # 常见的约束
    # NOT NULL 不为空  UNIQUE 不可重复  DEFAULT 设置默认值
    
    DESCRIBE `student`; # 描述一下表格
    ALTER TABLE `student` ADD `gpa` DECIMAL(3,2); # 给表格中添加一个属性
    ALTER TABLE `student` DROP COLUMN `gpa`; # 删除表格中的gpa属性
    DROP TABLE `student`;
    SELECT * FROM `student`;
    
    # 插入
    INSERT INTO `student` VALUES("小白",'历史');
    # 插入时指定属性的顺序
    INSERT INTO `student`(`name`,`major`) VALUES("小黑",'历史');
    
    
    # 修改、删除
    CREATE TABLE `student`(
        `student_id` INT PRIMARY KEY,
        `name` VARCHAR(50), # 约束 不为空
        `major` VARCHAR(50), # 不可重复
        `score` INT

    # 取得资料
    SELECT * FROM `student` ORDER BY `score` DESC LIMIT 2; # 排序 order by  desc 代表降序 limit 表示只取前2个数据
    SELECT `name` ,`major` FROM `student` ORDER BY `score`;
    
    
    
    
    
    
    # 创建公司数据库
    CREATE TABLE `employee`(
        `emp_id` INT PRIMARY KEY,
        `name` VARCHAR(20),
        `birth_data` DATE,
        `sex` VARCHAR(1),
        `salary` INT,
        `branch_id` INT,
        `sup_id` INT
    );
    
    create table `branch`(
        `branch_id` int primary key,
        `branch_name` varchar(20),
        `manager_id` int,
        foreign key (`manager_id`) references `employee`(`emp_id`) on delete set null
    );
    
    # 补充刚刚没有的foreign key
    alter table `employee` add foreign key (`sup_id`) references `employee`(`emp_id`) on delete set null;
    ALTER TABLE `employee` ADD FOREIGN KEY (`branch_id`) REFERENCES `branch`(`branch_id`) ON DELETE SET NULL;
    
    create table `client`(
        `client_id` INT primary key,
        `client_name` varchar(20),
        `phone` varchar(11)
    );
    CREATE TABLE `works_with`(
        `emp_id` INT ,
        `client_id` INT,
        `total_sales` int,
        primary key(`emp_id`,`client_id`),
        foreign key (`emp_id`) references `employee`(`emp_id`) on delete cascade,
        FOREIGN KEY (`client_id`) REFERENCES `client`(`client_id`) ON DELETE CASCADE
    );

    # 练习
    # 1. 取得所有员工资料
    select * from `employee`;
    # 2. 取得所有客户资料
    select * from `client`;
    # 3. 按薪水从低到高取得员工资料
    SELECT * FROM `employee` order by `salary`;
    # 4. 取得薪水前三高的员工
    SELECT * FROM `employee` ORDER BY `salary` desc limit 3;
    # 2. 取得所有员工的名字
    select `name` from `employee`;
    
    
    # aggregate function 聚合函数
    # 1.查看员工人数
    select count(*) from employee;
    # 2.取得所有出生于1970-01-01之后的女性员工人数
    SELECT COUNT(*) FROM `employee` where birth_data > '1970-01-01' and `sex` = 'F';
    # 3.取得所有员工的平均薪水
    select avg(`salary`) from `employee`;
    # 4.取得所有员工薪水的总和
    SELECT sum(`salary`) FROM `employee`;
    # 5.取得薪水最高的员工
    select max(`salary`) from `employee`;
    # 5.取得薪水最低的员工
    SELECT min(`salary`) FROM `employee`;
    
    
    # wildcards 通配符 %代表多个字，_代表一个字
    # 1.取得电话号码是335的客户
    select * from `client` where `phone` like '%335';
    
    # 取得姓艾的客户
    select * from `client` where `client_name` like '艾%';
    
    # 取得生日在12月的员工
    select * from `employee` where `birth_data` like '%-09-__';
    
    # union 合集,作用就是把多个结果集放在一起，要求列数相同而且数据类型兼容,列名由第一个查询决定
    # 1.员工名字和客户名字
    select `name` from `employee` union select `client_name` from `client`;
    # 员工id + 员工名字 union 客户id + 客户名字
    select `emp_id` as `total_id`,`name` as `total_name` from `employee` union select `client_id`,`client_name` from `client`;
    # 员工薪水 union 销售金额
    select `salary` from `employee` union select `total_sales` from `works_with`;
    
    
    # join 连接
    # 1.取得所有经理的名字
    select `employee`.`emp_id`,`employee`.`name`,`branch`.`branch_name` from `employee` join `branch` on `employee`.`emp_id` = `branch`.`manager_id`;
    
    # left join 不管右边成不成立，都会把左边表格的所有数据返回，不存在的值设为null
    select * from `employee` left join `branch` on `emp_id` = `manager_id`;
    
    # subquery 子查询
    # 1.找出研发部门的经理名字
    select `name` from `employee` where `emp_id` = (
        select `manager_id` from `branch` where `branch_name` = '研发'
    );
    
    # 2,找出对单位客户销售额超过5w的员工名字
    select * from `employee` where `emp_id` in (
        select `emp_id` from `works_with` where `total_sales` > 50000
    );

    # on delete cascade 就是跟着一起删掉

SQL231

alter和create 都可以创建索引。
Alter可以省略索引名。如果省略索引名，数据库会默认根据第一个索引列赋予一个名称；Create必须指定索引名称。

Create不能用于创建Primary key索引；

Alter允许一条语句同时创建多个索引；Create一次只能创建一个索引
ALTER TABLE 表名 ADD Primary key (col);
添加唯一索引
> 特点：索引列是唯一的，可以null；Mysql主动将该字段进行排序

    ALTER TABLE 表名 ADD unique <索引名> (col1, col2, ...col3);
    添加普通索引
    > 特点：添加普通索引， 索引值不唯一，可为null
    
    
    Alter table 表名 ADD index <索引名> (col1, col2, ...,);
    添加全文索引
    > 特点：只能在文本类型CHAR，VARCHAR, TEXT类型字段上创建全文索引；
    
    
    ALTER TABLE 表名 ADD Fulltext <索引名> (col)
    添加多列索引
    > 特点：多列是唯一的
    
    
    ALTER TABLE 表名 ADD UNIQUE (col1, col2, ..., )
    1.2 使用Create创建索引
    > 语法：create index 索引名 on 表名(字段)
    
    添加普通索引
    
    create index 索引名 on 表名(col1, col2, ..., )
    添加唯一索引
